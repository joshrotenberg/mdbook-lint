117    /// Load configuration from a file, auto-detecting format by extension
118    pub fn from_file<P: AsRef<Path>>(path: P) -> Result<Self> {
119        let path = path.as_ref();
120        let content = std::fs::read_to_string(path).map_err(|e| {
121            MdBookLintError::config_error(format!(
122                "Failed to read config file {}: {}",
123                path.display(),
124                e
125            ))
126        })?;
127
128        // Check if this is a markdownlint.json file
129        if path
130            .file_name()
131            .and_then(|name| name.to_str())
132            .map(|name| name == "markdownlint.json" || name == ".markdownlint.json")
133            .unwrap_or(false)
134        {
135            return Self::from_markdownlint_json(&content);
136        }
137
138        // Auto-detect format from file extension
139        match path.extension().and_then(|ext| ext.to_str()) {
140            Some("toml") => Self::from_toml_str(&content),
141            Some("yaml") | Some("yml") => Self::from_yaml_str(&content),
142            Some("json") => Self::from_json_str(&content),
143            _ => {
144                // Try to detect format from content if extension is ambiguous
145                Self::detect_format_from_content(&content).map_err(|e| {
146                    MdBookLintError::config_error(format!(
147                        "Unable to determine config format for {}: {}",
148                        path.display(),
149                        e
150                    ))
151                })
152            }
153        }
154    }

206    /// Convert markdownlint configuration format to our internal format
207    pub fn from_markdownlint_config(config: &serde_json::Value) -> Result<Self> {
208        let mut config_result = Config::default();
209        let mut rule_configs = HashMap::new();
210
211        if let Some(obj) = config.as_object() {
212            for (key, value) in obj {
213                // Skip comment fields
214                if key.starts_with("//") {
215                    continue;
216                }
217
218                match key.as_str() {
219                    // Global settings
220                    "default" => {
221                        if let Some(default_val) = value.as_bool() {
222                            if !default_val {
223                                // If default is false, start with empty enabled categories
224                                config_result.enabled_categories.clear();
225                            }
226                        }
227                    }
228
229                    // Map markdownlint rule names to MD### rule IDs
230                    "heading-style" => {
231                        if let Some(style_config) = Self::convert_markdownlint_rule(value, "MD003")
232                        {
233                            rule_configs.insert("MD003".to_string(), style_config);
234                        }
235                    }
236                    "heading-start-left" => {
237                        if value.as_bool() == Some(true) {
238                            // This is enforced by default in MD018
239                        }
240                    }
241                    "heading-increment" => {
242                        if value.as_bool() == Some(true) {
243                            // This is enforced by MD001
244                        }
245                    }
246                    "no-duplicate-heading" => {
247                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD024") {
248                            rule_configs.insert("MD024".to_string(), config);
249                        }
250                    }
251                    "first-heading-h1" => {
252                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD041") {
253                            rule_configs.insert("MD041".to_string(), config);
254                        }
255                    }
256                    "blanks-around-headings" => {
257                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD022") {
258                            rule_configs.insert("MD022".to_string(), config);
259                        }
260                    }
261                    "ul-style" => {
262                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD004") {
263                            rule_configs.insert("MD004".to_string(), config);
264                        }
265                    }
266                    "ul-indent" => {
267                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD007") {
268                            rule_configs.insert("MD007".to_string(), config);
269                        }
270                    }
271                    "ol-prefix" => {
272                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD029") {
273                            rule_configs.insert("MD029".to_string(), config);
274                        }
275                    }
276                    "line-length" => {
277                        if let Some(config) = Self::convert_line_length_config(value) {
278                            rule_configs.insert("MD013".to_string(), config);
279                        }
280                    }
281                    "no-hard-tabs" => {
282                        if value.as_bool() == Some(true) {
283                            // This is enforced by MD010
284                        }
285                    }
286                    "no-trailing-spaces" => {
287                        if value.as_bool() == Some(false) {
288                            config_result.disabled_rules.push("MD009".to_string());
289                        } else if let Some(config) = Self::convert_markdownlint_rule(value, "MD009")
290                        {
291                            rule_configs.insert("MD009".to_string(), config);
292                        }
293                    }
294                    "no-multiple-blanks" => {
295                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD012") {
296                            rule_configs.insert("MD012".to_string(), config);
297                        }
298                    }
299                    "code-block-style" => {
300                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD046") {
301                            rule_configs.insert("MD046".to_string(), config);
302                        }
303                    }
304                    "code-fence-style" => {
305                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD048") {
306                            rule_configs.insert("MD048".to_string(), config);
307                        }
308                    }
309                    "fenced-code-language" => {
310                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD040") {
311                            rule_configs.insert("MD040".to_string(), config);
312                        }
313                    }
314                    "no-inline-html" => {
315                        if let Some(config) = Self::convert_inline_html_config(value) {
316                            rule_configs.insert("MD033".to_string(), config);
317                        }
318                    }
319                    "emphasis-style" => {
320                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD049") {
321                            rule_configs.insert("MD049".to_string(), config);
322                        }
323                    }
324                    "strong-style" => {
325                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD050") {
326                            rule_configs.insert("MD050".to_string(), config);
327                        }
328                    }
329                    "hr-style" => {
330                        if let Some(config) = Self::convert_markdownlint_rule(value, "MD035") {
331                            rule_configs.insert("MD035".to_string(), config);
332                        }
333                    }
334                    _ => {
335                        // For unknown rules, check if it's a boolean disable
336                        if value.as_bool() == Some(false) {
337                            // Try to map the rule name to an MD### rule and disable it
338                            if let Some(md_rule) = Self::markdownlint_to_md_rule(key) {
339                                config_result.disabled_rules.push(md_rule);
340                            }
341                        }
342                    }
343                }
344            }
345        }
346
347        config_result.rule_configs = rule_configs;
348        Ok(config_result)
349    }

358    fn convert_line_length_config(value: &serde_json::Value) -> Option<toml::Value> {
359        if let Some(obj) = value.as_object() {
360            let mut config = toml::value::Table::new();
361
362            if let Some(length) = obj.get("line_length").and_then(|v| v.as_i64()) {
363                config.insert("line-length".to_string(), toml::Value::Integer(length));
364            }
365
366            if let Some(code_blocks) = obj.get("code_blocks").and_then(|v| v.as_bool()) {
367                config.insert(
368                    "ignore-code-blocks".to_string(),
369                    toml::Value::Boolean(!code_blocks),
370                );
371            }
372
373            if let Some(tables) = obj.get("tables").and_then(|v| v.as_bool()) {
374                config.insert("ignore-tables".to_string(), toml::Value::Boolean(!tables));
375            }
376
377            Some(toml::Value::Table(config))
378        } else {
379            None
380        }
381    }

436    /// Attempt to detect configuration format from content and parse
437    fn detect_format_from_content(content: &str) -> Result<Self> {
438        let trimmed = content.trim();
439
440        // Try JSON first (starts with { or [)
441        if trimmed.starts_with('{') || trimmed.starts_with('[') {
442            if let Ok(config) = Self::from_json_str(content) {
443                return Ok(config);
444            }
445        }
446
447        // Try YAML (often starts with --- or has key: value format)
448        if trimmed.starts_with("---") || content.contains(": ") {
449            if let Ok(config) = Self::from_yaml_str(content) {
450                return Ok(config);
451            }
452        }
453
454        // Fall back to TOML
455        Self::from_toml_str(content).map_err(|e| {
456            MdBookLintError::config_error(format!("Failed to parse as TOML, YAML, or JSON: {e}"))
457        })
458    }

460    /// Load configuration from mdBook preprocessor config
461    pub fn from_preprocessor_config(config: &serde_json::Value) -> Result<Self> {
462        let preprocessor_config = config
463            .get("preprocessor")
464            .and_then(|p| p.get("lint"))
465            .unwrap_or(&serde_json::Value::Null);
466
467        if preprocessor_config.is_null() {
468            return Ok(Self::default());
469        }
470
471        // Convert JSON to TOML-compatible format for parsing
472        let toml_value: toml::Value =
473            serde_json::from_value(preprocessor_config.clone()).map_err(|e| {
474                MdBookLintError::config_error(format!("Failed to convert preprocessor config: {e}"))
475            })?;
476
477        toml_value.try_into().map_err(|e| {
478            MdBookLintError::config_error(format!(
479                "Failed to parse preprocessor configuration: {e}"
480            ))
481        })
482    }

484    /// Check if a rule should be enabled based on configuration
485    pub fn should_run_rule(&self, rule_id: &str) -> bool {
486        // Explicit disabled rules always take precedence
487        if self.disabled_rules.contains(&rule_id.to_string()) {
488            return false;
489        }
490
491        // If we have explicit enabled rules, only those should run
492        if !self.enabled_rules.is_empty() {
493            return self.enabled_rules.contains(&rule_id.to_string());
494        }
495
496        // Check markdownlint compatibility mode - disable rules that are disabled by default in markdownlint
497        if self.markdownlint_compatible && rule_id == "MD044" {
498            return false; // proper-names: disabled by default in markdownlint
499        }
500
501        // Check category-based configuration
502        let category = self.get_rule_category(rule_id);
503        let category_name = self.category_to_string(&category);
504
505        if self.disabled_categories.contains(&category_name) {
506            return false;
507        }
508
509        // If enabled_categories is not empty, only run rules in enabled categories
510        if !self.enabled_categories.is_empty() {
511            return self.enabled_categories.contains(&category_name);
512        }
513
514        // Default: run all rules except those in disabled categories
515        true
516    }

518    /// Get the category for a rule based on its ID
519    fn get_rule_category(&self, rule_id: &str) -> RuleCategory {
520        match rule_id {
521            // Structure rules
522            "MD001" | "MD022" | "MD023" | "MD041" | "MD043" => RuleCategory::Structure,
523
524            // Style rules
525            "MD003" | "MD004" | "MD035" | "MD049" | "MD050" => RuleCategory::Style,
526
527            // Whitespace rules
528            "MD009" | "MD010" | "MD012" | "MD047" => RuleCategory::Whitespace,
529
530            // Code rules
531            "MD013" | "MD014" | "MD031" | "MD038" | "MD040" | "MD046" | "MD048" => {
532                RuleCategory::Code
533            }
534
535            // Link rules
536            "MD011" | "MD034" | "MD039" | "MD042" | "MD051" | "MD052" | "MD053" | "MD054" => {
537                RuleCategory::Links
538            }
539
540            // mdBook-specific rules
541            rule_id if rule_id.starts_with("MDBOOK") => RuleCategory::MdBook,
542
543            // Default to structure for unknown rules
544            _ => RuleCategory::Structure,
545        }
546    }

565    /// Merge this config with another, with the other taking precedence
566    pub fn merge(&mut self, other: Config) {
567        if other.fail_on_warnings {
568            self.fail_on_warnings = other.fail_on_warnings;
569        }
570        if !other.fail_on_errors {
571            self.fail_on_errors = other.fail_on_errors;
572        }
573        if other.markdownlint_compatible {
574            self.markdownlint_compatible = other.markdownlint_compatible;
575        }
576
577        // Merge rule lists
578        if !other.enabled_rules.is_empty() {
579            self.enabled_rules = other.enabled_rules;
580        }
581        if !other.disabled_rules.is_empty() {
582            self.disabled_rules = other.disabled_rules;
583        }
584        if !other.enabled_categories.is_empty() {
585            self.enabled_categories = other.enabled_categories;
586        }
587        if !other.disabled_categories.is_empty() {
588            self.disabled_categories = other.disabled_categories;
589        }
590
591        // Merge rule-specific configs
592        self.rule_configs.extend(other.rule_configs);
593    }

599    /// Auto-detect format and parse configuration from string
600    ///
601    /// Tries to parse the input as JSON, TOML, then YAML in that order.
602    fn from_str(content: &str) -> Result<Self> {
603        // Try JSON first (most structured)
604        if let Ok(config) = Self::from_json_str(content) {
605            return Ok(config);
606        }
607
608        // Try TOML next (common for Rust projects)
609        if let Ok(config) = Self::from_toml_str(content) {
610            return Ok(config);
611        }
612
613        // Try YAML last
614        if let Ok(config) = Self::from_yaml_str(content) {
615            return Ok(config);
616        }
617
618        Err(MdBookLintError::config_error(
619            "Could not parse configuration as JSON, TOML, or YAML",
620        ))
621    }
