34    /// Load configuration from preprocessor context
35    pub fn load_config_from_context(
36        &mut self,
37        ctx: &PreprocessorContext,
38    ) -> crate::error::Result<()> {
39        // Try both "mdbook-lint" and "lint" keys for backwards compatibility
40        let config = ctx
41            .config
42            .get_preprocessor("mdbook-lint")
43            .or_else(|| ctx.config.get_preprocessor("lint"));
44
45        if let Some(config) = config {
46            self.config = parse_mdbook_config(config)?;
47        }
48        Ok(())
49    }
50
51    /// Process a chapter and return any violations found
52    fn process_chapter(&self, chapter: &Chapter) -> crate::error::Result<Vec<crate::Violation>> {
53        // Create document from chapter content
54        let source_path = chapter
55            .source_path
56            .as_ref()
57            .unwrap_or(&PathBuf::from("unknown.md"))
58            .clone();
59
60        let document = Document::new(chapter.content.clone(), source_path)?;
61
62        // Use optimized checking (single AST parse) with configuration
63        let violations = self
64            .engine
65            .lint_document_with_config(&document, &self.config)?;
66
67        Ok(violations)
68    }
69
70    /// Format violations for output
71    fn format_violations(&self, violations: &[crate::Violation], chapter_path: &str) -> String {
72        if violations.is_empty() {
73            return String::new();
74        }
75
76        let mut output = String::new();
77        for violation in violations {
78            output.push_str(&format!(
79                "{}:{}:{}: {}\n",
80                chapter_path, violation.line, violation.column, violation
81            ));
82        }
83        output
84    }
85
86    /// Determine if we should fail the build based on violations
87    fn should_fail_build(&self, violations: &[crate::Violation]) -> bool {
88        for violation in violations {
89            match violation.severity {
90                Severity::Error if self.config.fail_on_errors => return true,
91                Severity::Warning if self.config.fail_on_warnings => return true,
92                _ => {}
93            }
94        }
95        false
96    }
97}

105impl Preprocessor for MdBookLint {
106    fn name(&self) -> &str {
107        "lint"
108    }
109
110    fn run(&self, _ctx: &PreprocessorContext, book: Book) -> mdbook::errors::Result<Book> {
111        let mut total_violations = Vec::new();
112        let mut should_fail = false;
113
114        // Process each chapter
115        for item in book.iter() {
116            if let BookItem::Chapter(chapter) = item {
117                let violations = self.process_chapter(chapter).map_err(|e| {
118                    mdbook::errors::Error::msg(format!("Failed to process chapter: {e}"))
119                })?;
120
121                if !violations.is_empty() {
122                    let chapter_path = chapter
123                        .source_path
124                        .as_ref()
125                        .map(|p| p.to_string_lossy())
126                        .unwrap_or("unknown".into());
127
128                    // Print violations to stderr
129                    eprint!("{}", self.format_violations(&violations, &chapter_path));
130
131                    if self.should_fail_build(&violations) {
132                        should_fail = true;
133                    }
134
135                    total_violations.extend(violations);
136                }
137            }
138        }
139
140        // Print summary
141        if !total_violations.is_empty() {
142            let error_count = total_violations
143                .iter()
144                .filter(|v| v.severity == Severity::Error)
145                .count();
146            let warning_count = total_violations
147                .iter()
148                .filter(|v| v.severity == Severity::Warning)
149                .count();
150            let info_count = total_violations
151                .iter()
152                .filter(|v| v.severity == Severity::Info)
153                .count();
154
155            eprintln!(
156                "mdbook-lint: {error_count} error(s), {warning_count} warning(s), {info_count} info"
157            );
158
159            if should_fail {
160                return Err(mdbook::errors::Error::msg(format!(
161                    "mdbook-lint: Build failed due to {error_count} error(s) and {warning_count} warning(s)"
162                )));
163            }
164        } else {
165            eprintln!("mdbook-lint: No issues found");
166        }
167
168        // Return the book unchanged (we're just linting, not modifying)
169        Ok(book)
170    }
171
172    fn supports_renderer(&self, renderer: &str) -> bool {
173        // We support all renderers since we don't modify content
174        match renderer {
175            "html" | "markdown" | "epub" | "pdf" => true,
176            _ => true, // Default to supporting unknown renderers
177        }
178    }

181/// Parse preprocessor configuration from mdbook config
182fn parse_mdbook_config(config: &toml::value::Table) -> crate::error::Result<Config> {
183    let mut preprocessor_config = Config::default();
184
185    if let Some(fail_on_warnings) = config.get("fail-on-warnings") {
186        preprocessor_config.fail_on_warnings = fail_on_warnings.as_bool().ok_or_else(|| {
187            crate::error::MdBookLintError::config_error("fail-on-warnings must be a boolean")
188        })?;
189    }
190
191    if let Some(fail_on_errors) = config.get("fail-on-errors") {
192        preprocessor_config.fail_on_errors = fail_on_errors.as_bool().ok_or_else(|| {
193            crate::error::MdBookLintError::config_error("fail-on-errors must be a boolean")
194        })?;
195    }
196
197    if let Some(enabled_categories) = config.get("enabled-categories") {
198        if let Some(categories_array) = enabled_categories.as_array() {
199            for category in categories_array {
200                if let Some(category_str) = category.as_str() {
201                    preprocessor_config
202                        .enabled_categories
203                        .push(category_str.to_string());
204                }
205            }
206        }
207    }
208
209    if let Some(disabled_categories) = config.get("disabled-categories") {
210        if let Some(categories_array) = disabled_categories.as_array() {
211            for category in categories_array {
212                if let Some(category_str) = category.as_str() {
213                    preprocessor_config
214                        .disabled_categories
215                        .push(category_str.to_string());
216                }
217            }
218        }
219    }
220
221    if let Some(enabled_rules) = config.get("enabled-rules") {
222        if let Some(rules_array) = enabled_rules.as_array() {
223            for rule in rules_array {
224                if let Some(rule_str) = rule.as_str() {
225                    preprocessor_config.enabled_rules.push(rule_str.to_string());
226                }
227            }
228        }
229    }
230
231    if let Some(disabled_rules) = config.get("disabled-rules") {
232        if let Some(rules_array) = disabled_rules.as_array() {
233            for rule in rules_array {
234                if let Some(rule_str) = rule.as_str() {
235                    preprocessor_config
236                        .disabled_rules
237                        .push(rule_str.to_string());
238                }
239            }
240        }
241    }
242
243    Ok(preprocessor_config)
244}

312/// Handle the preprocessor protocol (stdin/stdout communication with mdbook)
313pub fn handle_preprocessing() -> crate::error::Result<()> {
314    let mut stdin = io::stdin();
315    let mut input = String::new();
316    stdin
317        .read_to_string(&mut input)
318        .map_err(crate::error::MdBookLintError::Io)?;
319
320    let (ctx, book): (PreprocessorContext, Book) =
321        serde_json::from_str(&input).map_err(crate::error::MdBookLintError::Json)?;
322
323    let mut preprocessor = MdBookLint::new();
324    preprocessor.load_config_from_context(&ctx)?;
325
326    let processed_book = preprocessor.run(&ctx, book).map_err(|e| {
327        crate::error::MdBookLintError::document_error(format!("Preprocessor failed: {e}"))
328    })?;
329
330    let output =
331        serde_json::to_string(&processed_book).map_err(crate::error::MdBookLintError::Json)?;
332
333    print!("{output}");
334    Ok(())
335}
